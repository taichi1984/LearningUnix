# プログラミング課題12.8
## 課題内容
**HEADメソッド** HEADリクエストをサポートするように、Webサーバーを書き換えなさい。詳細はHTTPの仕様書を参照しなさい。

## HEADメソッドとは
RFC9110には以下のような記載がある。

```
HEADメソッドは、サーバーが応答にコンテンツを送信してはならないことを除いて、GETするのと同じです。
HEADは、多くの場合、ハイパーテキストリンクをテストしたり、最近の変更を見つけたりするために、表現データを転送せずに選択した表現に関するメタデータを取得するために使用されます。
```

と書いてある。
要はヘッダだけ送信すれば良い。

## 現在はどのようにリクエストを処理しているか。
現時点ではGETのみを対応するようにしている。
この内容をHEADやPOSTなどの他のリクエストであっても、リクエストを判別し、処理を分岐できるようにしなければならない。
今回はHEADリクエストだった場合、bodyをsendしないように条件を管理すればよいだろう。

void process_rq(char *rq, int fd, struct http_response *response, char *addr) {

  strcpy(arg, "./");

    return;

    int handle = 0;
    if (strcmp(cmd, "GET") != 0) {
      cannot_do(response);
    } else if (not_exist(arg)) {
      do_404(arg, response);
    } else if (isadir(arg)) {
      do_ls(arg, fd, response);
      handle = 1;
    } else if (in_cgi_bin(arg) || ends_in_cgi(arg)) {
      initialize_env(cmd, addr);
      do_exec(arg, fd);

      handle = 1;

      send_header(fp, &response->header);
      send_body(fp, response);
      fflush(fp);
      fclose(fp);
      _exit(1);

  }

  }

  fprintf(fp, "\r\n");
}

  response->header.date = time(NULL);

  if (stat(f, &st) == -1) {
    perror("stat");
    do_404(f, response);
    return -1;
  }
  response->header.last_modified = st.st_mtime;
  response->header.content_length = st.st_size;
  response->header.accept_ranges = 1;

    fclose(fpfile);
    fclose(fpsock);

  _exit(0);
}

現行ではrequestという文字列が命令の種類を持っているが、これを上手く使って処理を分岐させたい。
requestという文字列が正しいかどうかなどを判断させた上で、状態としてrequestの種類を持つのが良いだろう。
request文字列の他に、request構造体を作り、requestに書かれている内容を保持するのがいい方法だと思われる。
本課題では、do_lsとdo_execの対応は難易度が高いため、do_lsはcontent_lengthなどを表示せず、do_execは対応しない。
do_catやdo_404を完全に対応するものとする。

void process_rq(struct http_request *request, int fd,
                struct http_response *response, char *addr) {

  if (fork() == 0) {
    int handle = 0;
    if (request->method == METHOD_UNKNOWN) {
      cannot_do(response);
    } else if (not_exist(request->path)) {
      do_404(request->path, response);
    } else if (isadir(request->path)) {
      do_ls(request->path, fd, request, response);
      handle = 1;
    } else if (in_cgi_bin(request->path) || ends_in_cgi(request->path)) {
      initialize_env(request->method_str, addr);
      do_exec(request->path, fd);
      handle = 1;
    } else {
      do_cat(request->path, fd, request, response);
      handle = 1;
    }

    if (!handle) {
      FILE *fp = fdopen(fd, "w");
      send_header(fp, &response->header);
      if (request->method != METHOD_HEAD)
        send_body(fp, response);
      fflush(fp);
      fclose(fp);
      _exit(1);
    }
  }
}

void send_header(FILE *fp, struct http_header *h) {

    if (h->charset)
      fprintf(fp, "; charset=%s", h->charset);
    fprintf(fp, "\r\n");
  }
  if (h->last_modified) {
    struct tm *tm;
    char buf[128];
    tm = gmtime(&h->last_modified);

  response->header.content_type = "text/plain";
  send_header(fp, &response->header);

  fflush(fp);

  if (request->method != METHOD_HEAD) {
    dup2(fd, 1);
    dup2(fd, 2);
    close(fd);

    perror(dir);
  }
  exit(1);
}

/* --------------------------------------------*
 * do_cat(filename , fd)
 * ヘッダを送ってから内容を送り返す
 * --------------------------------------------*/

int do_cat(char *f, int fd, struct http_request *request,
           struct http_response *response) {
  char *extension = file_type(f);
  struct stat st;
  char timebuf[128];
  time_t now = time(NULL);
  struct tm *tm = gmtime(&now);

  response->header.content_type = "text/plain";

  FILE *fpsock, *fpfile;
  int c;
  response->header.http_version = "HTTP/1.0";
  response->header.status_code = 200;
  response->header.reason_phrase = "OK";
  response->header.server = "MyWebServer 1.0";
  if (strcmp(extension, "html") == 0)
    response->header.content_type = "text/html";
  else if (strcmp(extension, "gif") == 0)
    response->header.content_type = "image/gif";
  else if (strcmp(extension, "jpg") == 0)
    response->header.content_type = "image/jpeg";
  else if (strcmp(extension, "jpeg") == 0)
    response->header.content_type = "image/jpeg";

  response->header.charset = "UTF-8";
  response->header.date = time(NULL);

  if (stat(f, &st) == -1) {
    perror("stat");
    do_404(f, response);
    return -1;
  }
  response->header.last_modified = st.st_mtime;
  response->header.content_length = st.st_size;
  response->header.accept_ranges = 1;
  response->header.connection_close = 1;

  fpsock = fdopen(fd, "w");
  fpfile = fopen(f, "r");
  if (fpsock != NULL && fpfile != NULL) {
    send_header(fpsock, &response->header);
    if (request->method != METHOD_HEAD)
      while ((c = getc(fpfile)) != EOF)
        putc(c, fpsock);

    fclose(fpfile);
    fclose(fpsock);
  }
  _exit(0);
}

int initialize_env(const char *cmd, const char *addr) {
  int rv = -1;
  if ((rv = setenv("REQUEST_METHOD", cmd, 1)) == -1) {
    perror("setenv");
    return rv;
  }
  if ((rv = setenv("REMOTE_ADDR", addr, 1)) == -1) {
    perror("setenv");
    return rv;
  }

  return rv;
}

}

enum http_method parse_method(const char *s) {

  if (strcmp(s, "GET") == 0)

  if (strcmp(s, "HEAD") == 0)

  if (strcmp(s, "CONNECT") == 0)

}
