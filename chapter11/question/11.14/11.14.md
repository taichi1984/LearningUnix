# プログラミング課題11.14
## 課題内容
**fingerサーバー** ポート79で実行されているのを本文でも紹介した個人情報検索サーバーの独自バージョンを書きなさい。
サーバは、１行の入力という形でユーザー名を受付、ローカルユーザーリストでそれにマッチしたレコードのリストをクライアントに送り返すものとする。

どうやらユーザーの情報を色々表示している。

## /etc/passwdの内容を取得するにはどうすればいいか？
ファイルを読んで自前で処理するのも可能だが、幸いなことにpasswdは構造体で指定したユーザーを取得できるlibcが用意されている。
struct passwd *getpwnam(const char *name) 関数がそれに当たる。

この関数は成功すると指定したユーザーのpasswd構造体を返す。
ユーザーがいない場合や失敗した場合はNULLを返すようにできている。
ユーザーがいない場合はerrnoは更新されないが失敗した場合は更新される。
errnoを0に初期化してからgetpwnamを呼び出すことで、
NULLが返った場合にユーザー未存在かエラーかを判別できる。


passwd構造体は以下のようになっている。

```          
            struct passwd {
               char   *pw_name;       /* ユーザー名 */
               char   *pw_passwd;     /* ユーザーのパスワード */
               uid_t   pw_uid;        /* ユーザー ID */
               gid_t   pw_gid;        /* グループ ID */
               char   *pw_gecos;      /* ユーザー情報 */
               char   *pw_dir;        /* ホームディレクトリ */
               char   *pw_shell;      /* シェルプログラム */
           };
```

## コード
```

#include <alloca.h>
#include <netdb.h>
#include <netinet/in.h>
#include <pwd.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>
#include <unistd.h>

#define PORTNUM 7979
#define HOSTLEN 256
int setup_server(int port);

int main() {
  int sock_id, sock_fd;
  FILE *sock_fp;

  sock_id = setup_server(PORTNUM);
  if (sock_id == -1) {
    perror("socket");
    exit(1);
  }

  /* メインループ*/

  while (1) {

    char request[BUFSIZ];
    sock_fd = accept(sock_id, NULL, NULL);
    if (sock_fd == -1) {
      perror("accept");
      continue;
    }
    sock_fp = fdopen(sock_fd, "r");
    if (sock_fp == NULL) {
      perror("fdopen");
      close(sock_fd);
      continue;
    }

    if (fgets(request, sizeof(request), sock_fp) == NULL) {
      perror("fgets");
      fclose(sock_fp);
      continue;
    };

    request[strcspn(request, "\r\n")] = '\0';

    struct passwd *pw = getpwnam(request);

    char buf[BUFSIZ];
    int len;

    if (pw == NULL) {
      len = snprintf(buf, sizeof(buf), "user: %s is not found \n", request);
    } else {
      len = snprintf(
          buf, sizeof(buf),
          "username : %s \n uid : %d \n gid : %d \n home : %s \n shell : %s \n",
          pw->pw_name, pw->pw_uid, pw->pw_gid, pw->pw_dir, pw->pw_shell);
    }

    write(sock_fd, buf, len); //部分書き込みになる可能性があるが、今回は簡略化している。

    fclose(sock_fp);
  }
  return 0;
}

int setup_server(int port) {
  struct sockaddr_in saddr;
  char hostname[HOSTLEN];
  int sock_id;

  /* ステップ1:カーネルにソケットを要求する。*/

  sock_id = socket(PF_INET, SOCK_STREAM, 0); // ソケット取得
  if (sock_id == -1)
    return -1;

  /* ステップ２：ソケットにアドレスをバインドする。アドレスはホスト、ポート*/

  memset((void *)&saddr, 0, sizeof(saddr)); /*構造体をクリアする*/

  saddr.sin_addr.s_addr = htonl(INADDR_ANY);
  saddr.sin_port = htons(
      PORTNUM); // htons()関数でリトルエンディアン→ビッグエンディアンに変換する。（ネットワークはビッグエンディアンが共通語になっている。）
  saddr.sin_family = AF_INET;

  if (bind(sock_id, (struct sockaddr *)&saddr, sizeof(saddr)) != 0)
    return -1;

  /*
   * ステップ3:キューサイズを１としてソケットを着信できる状態にする。
   */

  if (listen(sock_id, 5) != 0)
    return -1;

  return sock_id;
}

```

## まとめ
ソケット上での通信のやり取りはいつもどおりだが、今回は１行でユーザー名を取得できればいいので、
fgets(buf,sizeof(buf),sock_fp)を１回だけ行えばよい。
その情報を元にユーザーのpasswd情報を探すが、その際は何でも自力でファイルを読むのではなく、
今回のようにglibcに簡単に取得できるライブラリがある可能性があるので、便利なものがないか確認する癖をつけよう。
自前で実装するより、信頼できるライブラリを使ったほうが実現が簡単で可読性もよく、信頼性の高いプログラムを書ける。
