# プログラミング課題11.06
## 課題内容
tinybcに-cオプションを追加しなさい。このオプションを追加すると、次のコマンドが機能するはずである。

```
printf "2 + 2\n4 * 4\n" | tinybc -c | dc
```

## 現状のtinybcについて
### ソースコード
```
/** tinybc.c * dcを使って仕事をする簡単な電卓
 *　　　　　　* 双方向パイプの作り方、使い方を示す。
 *            * 入力は、数値、演算子、数値の順番で指定する。
 *            * tinybcは、これを数値、\n、数値、\n,演算子,
 *            \n , p に変換してdcに渡す。
 *            * 返された結果は、標準出力に書き込む。
 *
 *            *プログラムの概要
 *            *a. 2本のパイプを作る
 *            *b. fork呼び出し（新しいプロセスを得る）
 *            *c.
 *dcになる予定の新プロセスの中で、標準入出力をパイプに接続し、execl dcする。 *d.
 *tinybcプロセスの中では、特に配管の必要はない。
 *
 *            通常の入出力を介して、ユーザーとやり取りし、
 *            パイプを介して、プログラム感の情報のやり取りをする。
 *            最後にパイプをクローズすると、dcは終了する。
 *            *注意:複数行のレスポンスを処理しない。
 *
 */

#include <stdio.h>
#include <stdlib.h>
#include <sys/wait.h>
#include <unistd.h>

#define oops(m, x)                                                             \
  {                                                                            \
    perror(m);                                                                 \
    exit(x);                                                                   \
  }
void be_dc(int in[2], int out[2]);
void be_bc(int todc[2], int fromdc[2]);
void fatal(char *mess);

int main() {
  int pid, todc[2], fromdc[2];

  /* 2本のパイプを作る */

  if (pipe(todc) == -1 || pipe(fromdc) == -1)
    oops("pipe failed", 1);

  /* ユーザーインターフェースのためのプロセスを作る */

  if ((pid = fork()) == -1)
    oops("cannot fork", 2);
  if (pid == 0)
    be_dc(todc, fromdc);
  else {
    be_bc(todc, fromdc);
    wait(NULL);
  }
}

void be_dc(int in[2], int out[2])
/*
 * 標準入出力をセットアップしてから、最後にexecl dc
 */
{
  /* 標準入力を入力用パイプに接続する*/
  if (dup2(in[0], 0) == -1)
    oops("dc: cannot redirect stdin", 3);
  close(in[0]);
  close(in[1]);

  /*標準出力を出力用パイプに接続する*/
  if (dup2(out[1], 1) == -1)
    oops("dc:cannot redirect stdout", 4);
  close(out[1]);
  close(out[0]);

  /* -オプションをつけてdcをexeclにする */
  execlp("dc", "dc", "-", NULL);
  oops("Cannot run dc", 5);
}

void be_bc(int todc[2], int fromdc[2])
/*
 * 標準入力から世j見出したものを逆ポーランド記法に変換し、パイプに送る。
 * 別のパイプから処理結果を読み出し、ユーザーにしめす。
 * fdopen()を呼び出して、ファイルディスクリプタをストリームに変換する。
 */
{
  int num1, num2;
  char operation[BUFSIZ], message[BUFSIZ];
  FILE *fpout, *fpin;

  /* セットアップ */
  close(todc[0]);
  close(fromdc[1]);

  fpout = fdopen(todc[1], "w");
  fpin = fdopen(fromdc[0], "r");

  if (fpout == NULL || fpin == NULL)
    fatal("Error convering pipes to streams");

  /* メインループ */
  while (1) {

    printf("tinybc : ");
    if (fgets(message, BUFSIZ, stdin) == NULL)
      break;

    /* parse input */
    if (sscanf(message, "%d%1[-+*/^]%d", &num1, operation, &num2) != 3) {
      printf("syntax error\n");
      continue;
    }

    if (fprintf(fpout, "%d\n%d\n%c\np\n", num1, num2, *operation) == EOF)

      fatal("Error writing");

    fflush(fpout);

    if (fgets(message, BUFSIZ, fpin) == NULL)
      break;

    printf("%d %c %d = %s", num1, *operation, num2, message);
  }
  fclose(fpout);
  fclose(fpin);
}

void fatal(char *mess) {
  fprintf(stderr, "Error: %s\n", mess);
  exit(1);
}

```

実行例を見る限り、できていないことは２つ。
まず１つ目は2 + 2　4 * 4のように、数字とオペランドの間にスペースが入ったときにスペースを無視することができていないようだ。
２つ目は、現状のtinybcは自前でdcにデータを送り、標準出力に計算の結果を出すため、パイプでdcにつなぐと、出した結果がdcに接続されてしまうのでうまく動かないようだ。
-cオプションをつけた場合には、tinybcはforkしてdcをexecしてそこにパイプをつなげるのではなく、素直に標準出力にbcで加工した文字列を出す必要がある。

空白の問題はオプションの有無とは独立した入力仕様の問題であるため、-cオプションとは切り離し、常に有効とすることにする。
-c オプションは、出力方式の変更のみに絞り、実装する。

## スペースを無視するにはどうすればいいか？
sscanfのフォーマットを修正で良い。
具体的には各フォーマット指定子の間にスペースを入れることで対処できる。
各フォーマット指定子の間にスペースを入れることで、任意の数のスペース(0個のスペースも含む)を各フォーマット指定子の間に挟んでも
各変数に正しく代入できるようになる。

## オプションごとにパイプをつなぐか判断するのを考える。
パイプを用意する場所、forkする場所など各所にif(!opt.flag_c)を入れるとソースコードの各所にif文が散らばる上
各関数にオプションを渡す必要性ができてしまうため可読性、メンテナンス性が大幅に落ちる。
この問題を解決するために、be_bc_stdoutという、標準出力に出力するようの関数を作成し、main関数内で処理を分岐させた。

```
if (opt.flag_c) {
    be_bc_stdout();
  } else {
    /* 2本のパイプを作る */
    if (pipe(todc) == -1 || pipe(fromdc) == -1)
      oops("pipe failed", 1);

    /* ユーザーインターフェースのためのプロセスを作る */

    if ((pid = fork()) == -1)
      oops("cannot fork", 2);
    if (pid == 0)
      be_dc(todc, fromdc);
    else {
      be_bc(todc, fromdc);
      wait(NULL);
    }
  }
```

## 改良版
### ソースコード
```
/** tinybc.c * dcを使って仕事をする簡単な電卓
 *　　　　　　* 双方向パイプの作り方、使い方を示す。
 *            * 入力は、数値、演算子、数値の順番で指定する。
 *            * tinybcは、これを数値、\n、数値、\n,演算子,
 *            \n , p に変換してdcに渡す。
 *            * 返された結果は、標準出力に書き込む。
 *
 *            *プログラムの概要
 *            *a. 2本のパイプを作る
 *            *b. fork呼び出し（新しいプロセスを得る）
 *            *c.
 *dcになる予定の新プロセスの中で、標準入出力をパイプに接続し、execl dcする。 *d.
 *tinybcプロセスの中では、特に配管の必要はない。
 *
 *            通常の入出力を介して、ユーザーとやり取りし、
 *            パイプを介して、プログラム感の情報のやり取りをする。
 *            最後にパイプをクローズすると、dcは終了する。
 *            *注意:複数行のレスポンスを処理しない。
 *
 */

#include <getopt.h>
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/wait.h>
#include <unistd.h>

#define oops(m, x)                                                             \
  {                                                                            \
    perror(m);                                                                 \
    exit(x);                                                                   \
  }
typedef struct {
  bool flag_c;
} Option;

void be_dc(int in[2], int out[2]);
void be_bc(int todc[2], int fromdc[2]);
void be_bc_stdout();
void fatal(char *mess);

int main(int ac, char *av[]) {
  Option opt = {0};
  int c;
  int pid, todc[2], fromdc[2];

  pid = getpid();
  while ((c = getopt(ac, av, "c")) != -1) {
    switch (c) {
    case 'c':
      opt.flag_c = true;
      break;
    case '?':
      exit(1);
    }
  }

  if (opt.flag_c) {
    be_bc_stdout();
  } else {
    /* 2本のパイプを作る */
    if (pipe(todc) == -1 || pipe(fromdc) == -1)
      oops("pipe failed", 1);

    /* ユーザーインターフェースのためのプロセスを作る */

    if ((pid = fork()) == -1)
      oops("cannot fork", 2);
    if (pid == 0)
      be_dc(todc, fromdc);
    else {
      be_bc(todc, fromdc);
      wait(NULL);
    }
  }
}

void be_dc(int in[2], int out[2])
/*
 * 標準入出力をセットアップしてから、最後にexecl dc
 */
{
  /* 標準入力を入力用パイプに接続する*/
  if (dup2(in[0], 0) == -1)
    oops("dc: cannot redirect stdin", 3);
  close(in[0]);
  close(in[1]);

  /*標準出力を出力用パイプに接続する*/
  if (dup2(out[1], 1) == -1)
    oops("dc:cannot redirect stdout", 4);
  close(out[1]);
  close(out[0]);

  /* -オプションをつけてdcをexeclにする */
  execlp("dc", "dc", "-", NULL);
  oops("Cannot run dc", 5);
}

void be_bc(int todc[2], int fromdc[2])
/*
 * 標準入力から世j見出したものを逆ポーランド記法に変換し、パイプに送る。
 * 別のパイプから処理結果を読み出し、ユーザーにしめす。
 * fdopen()を呼び出して、ファイルディスクリプタをストリームに変換する。
 */
{
  int num1, num2;
  char operation[2], message[BUFSIZ];
  FILE *fpout, *fpin;

  /* セットアップ */
  close(todc[0]);
  close(fromdc[1]);

  fpout = fdopen(todc[1], "w");
  fpin = fdopen(fromdc[0], "r");

  if (fpout == NULL || fpin == NULL)
    fatal("Error convering pipes to streams");

  /* メインループ */
  while (1) {
    if (isatty(0))
      printf("tinybc : ");

    if (fgets(message, BUFSIZ, stdin) == NULL)
      break;

    /* parse input */
    if (sscanf(message, "%d %1[-+*/^] %d", &num1, operation, &num2) != 3) {
      printf("syntax error\n");
      continue;
    }

    if (fprintf(fpout, "%d\n%d\n%c\np\n", num1, num2, *operation) == EOF)
      fatal("Error writing");

    fflush(fpout);

    if (fgets(message, BUFSIZ, fpin) == NULL)
      break;
    printf("%d %c %d = %s", num1, *operation, num2, message);
  }
  fclose(fpout);
  fclose(fpin);
}

void be_bc_stdout()
/*
 * 標準入力から読み出したものを逆ポーランド記法に変換し、標準出力に送る。
 */
{
  int num1, num2;
  char operation[2], message[BUFSIZ];

  /* メインループ */
  while (1) {
    if (fgets(message, BUFSIZ, stdin) == NULL)
      break;

    /* parse input */
    if (sscanf(message, "%d %1[-+*/^] %d", &num1, operation, &num2) != 3) {
      printf("syntax error\n");
      continue;
    }

    printf("%d\n%d\n%c\np\n", num1, num2, *operation);
    fflush(stdout);
  }
}

void fatal(char *mess) {
  fprintf(stderr, "Error: %s\n", mess);
  exit(1);
}
```

## まとめ
オプションなどで、pipeをつないだり繋がなかったりなど処理を分岐させたい場合は
処理ごとの関数を作り、メインで分岐させることで、分岐処理を１箇所にまとめる設計にした。
分岐処理が散らばると変更時に修正漏れが起きやすくなるので、重要ポイントである。
また、モードごとの責務がソースコード上で曖昧になる（非常に読み取りづらくなる）のも回避できる。

これにより可読性、メンテナンス性ともに良いコードに仕上がった。


## 今回試みたこと
### getopt関数の導入
main関数のオプションおよび引数の処理については、ほぼすべてのプログラムで必要となることから、
今までやってた自力での引数管理から、getopt関数を用いた管理に変更した。

### stdboolの導入
フラグなどの真偽値のみで判断する値に関してはstdboolを導入し、可読性をあげることにした。
関数の戻り値が３つ以上ある場合は、true(成功) false(失敗) -1（例外）などの無理な利用はせず、
真偽値にするにふさわしいものだけに適用している。

### isattyの導入
対話モードとパイプモードで出力を分けるためにisatty関数を利用した。
これにより、パイプ接続時に不要な出力が混入するのを防ぐことができた。
