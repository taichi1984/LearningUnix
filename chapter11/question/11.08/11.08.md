# プログラミング課題11.08
## 課題内容
pcloseを書きなさい。この関数は、引数としてpopenから返されたFILE *を取る。
fdopen関数は、バッファと細かい管理作業のためのメモリを確保している。
fclose関数は、そのメモリを開放し、下にあるファイルディスクリプタをクローズする。
pcloseは他に何をしなければならないだろうか。popen呼び出しからpclose呼び出しまでの間に他の子プロセスが強制終了したらどうすればよいか。

## そもそもpopenがどんな関数だったか。
popenがどんな関数だったかコードを確認する。

### コード
```
/* popen.c - UNIXライブラリ関数のpopen()の実装の１つ
 * FILE *popen(char *command , char *mode)
 * command は通常のシェルコマンド
 * modeは"r"か"w"
 * コマンドに接続されたストリームかNULを返す。
 * execlで"sh" "-c" command を実行する。
 * 課題：子プロセスのシグナル処理をどうするか。
 */

#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

#define READ 0
#define WRITE 1

FILE *popen(const char *command, const char *mode) {
  int pfp[2], pid;
  FILE *fp;
  int parenend, child_end;

  if (*mode == 'r') {
    parent_end = READ;
    child_end = WRITE;
  } else if (*mode == 'w') {
    parent_end = WRITE;
    child_end = READ;
  } else
    return NULL;

  if (pipe(pfp) == -1)
    return NULL;
  if ((pid = fork()) == -1) {
    close(pfp[0]);
    close(pfp[1]);
    return NULL;
  }

  /* ------------------- 親のコード -------------*/
  /* ------------------- 片方をクローズしてもう片方をfdopenする。*/

  if (pid > 0) {
    if (close(pfp[child_end]) == -1)
      return NULL;
    return fdopen(pfp[parent_end], mode);
  }

  /* -------------------子のコード ------------------*/
  /* stdin またはstdoutをリダイレクトし、cmdをexec */
  if (close(pfp[parent_end]) == -1)
    exit(1);

  if (dup2(pfp[child_end], child_end) == -1)
    exit(1);

  if (close(pfp[child_end] == -1))
    exit(1);

  execl("/bin/sh", "sh", "-c", command, NULL);
  exit(1);
}

```

子プロセスのほうはforkしたらパイプをつないで、sh -cを実行し、終了しているのみなので特にすることはない。
親プロセスは何をしているかというと、php[chile_end]側はcloseしているが、
return fdopen(pfp[parent_end],mode);

とパイプの片側をcloseせず、そのまま渡している。

pcloseはこれを閉じる必要がある。

## この内容をもとに自作のpcloseでやらなければならないことを考える。
pcloseは以下のことをしなければならない。

1,popenで作られたプロセスの終了を待つ。
2,引数で取得したfile pointerをcloseする。
3,pcloseの戻り値としてwait4で返されたコマンドの終了状態を返すようにする。

### popenからpcloseまでの間に他の子プロセスが強制終了した場合、どうするか？
pclose内でwait()を使った場合、他のプロセスが強制終了したら、wait()はその関係ない子プロセスの終了を持ってwaitをやめ、情報を返してしまう。
pclose内でwait4()するプロセスはpopenでforkしたプロセスである必要がある。

wait4の第１引数で何を指定するか？が重要になる。

```
pid に指定できる値は以下の通り:

< -1   プロセスグループID  が pid の絶対値に等しい子プロセスのいずれかが終了するまでを待つ。

-1     子プロセスのどれかが終了するまで待つ。

0      プロセスグループ ID が、waitpid()  が呼ばれた時点での呼び出し元のプロセスのプロセスグループ ID と等しい子プロセスを待つ。

> 0    プロセスID が pid に等しい子プロセスを待つ。
```

-1と >0の内容は簡単だ。
"-1" は最初に終了する子プロセスを待つ。これは他の子プロセスが終了したら意図せずwaitをやめてしまうので、今回は使えないだろう。

"> 0" の場合はプロセスIDがpidに等しい子プロセスを待つ。
これはpopenがプロセスIDを返す場合、実現可能であるが、popenはfile pointerを返す仕様になっており、引数でも情報を渡すような仕組みがないため、これ単体での実現は難しい。
これを実現させるためには、fpで繋がっているプロセスからpidを求めることができる機能の存在が必要だ。そのような機能はあるのだろうか？

結論からいうと、fpで繋がっているプロセスからpidを求めることができるような関数は存在しない。
そのため、popenでのfork時にpidを保存しておく変数を持っておく必要がある。


## 上記を踏まえ、popen pcloseを実装する。
pid保持の関係で、popenへの変更や、そのまわりに関数や変数の追加が必要になったためpopenも含めた全てのソースコードを掲載する。

```
/* popen.c - UNIXライブラリ関数のpopen()の実装の１つ
 * FILE *popen(char *command , char *mode)
 * command は通常のシェルコマンド
 * modeは"r"か"w"
 * コマンドに接続されたストリームかNULを返す。
 * execlで"sh" "-c" command を実行する。
 * 課題：子プロセスのシグナル処理をどうするか。
 */

#include <alloca.h>
#include <errno.h>
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/wait.h>
#include <unistd.h>

#define READ 0
#define WRITE 1
#define MAX_TABLE 255
typedef struct {
  FILE *fp;
  pid_t pid;
} Process;

static Process proc_table[MAX_TABLE];
int regist_process(Process p);
int remove_process(FILE *fp);

FILE *popen(const char *command, const char *mode) {
  int pfp[2], pid;
  int parent_end, child_end;

  if (*mode == 'r') {
    parent_end = READ;
    child_end = WRITE;
  } else if (*mode == 'w') {
    parent_end = WRITE;
    child_end = READ;
  } else
    return NULL;

  if (pipe(pfp) == -1)
    return NULL;
  if ((pid = fork()) == -1) {
    close(pfp[0]);
    close(pfp[1]);
    return NULL;
  }

  if (pid == 0) {
    /* -------------------子のコード ------------------*/
    /* stdin またはstdoutをリダイレクトし、cmdをexec */

    if (close(pfp[parent_end]) == -1)
      _exit(1);

    if (dup2(pfp[child_end], child_end) == -1)
      _exit(1);

    if (close(pfp[child_end]) == -1)
      _exit(1);

    execl("/bin/sh", "sh", "-c", command, NULL);
    _exit(1);
  } else {

    /* ------------------- 親のコード -------------*/
    /* ------------------- 片方をクローズしてもう片方をfdopenする。*/

    Process p;
    if (close(pfp[child_end]) == -1) {
      close(pfp[parent_end]);
      return NULL;
    }

    p.fp = fdopen(pfp[parent_end], mode);
    if (p.fp == NULL) {
      close(pfp[parent_end]);
      return NULL;
    }

    p.pid = pid;

    if (regist_process(p) == -1) {
      fclose(p.fp);
      waitpid(pid, NULL, 0); // 子プロセス回収
      return NULL;
    };

    return p.fp;
  }
}

int pclose(FILE *fp) {
  pid_t pid;
  int wstatus;

  if ((pid = remove_process(fp)) == -1) {
    return -1;
  }
  if (fclose(fp) == -1) {
    return -1;
  }

  while ((wait4(pid, &wstatus, 0, NULL)) == -1) {
    if (errno != EINTR)
      return -1;
  }

  return wstatus;
}

int regist_process(Process p) {
  for (int i = 0; i < MAX_TABLE; i++) {
    if (proc_table[i].fp == NULL) {
      proc_table[i].fp = p.fp;
      proc_table[i].pid = p.pid;
      return 0;
    }
  }
  return -1;
}

pid_t remove_process(FILE *fp) {
  pid_t pid = -1;
  for (int i = 0; i < MAX_TABLE; i++) {
    if (proc_table[i].fp == fp) {
      pid = proc_table[i].pid;
      proc_table[i].fp = NULL;
      return pid;
    }
  }

  return -1;
}

```
## まとめ
pcloseが行っていることは以下の３つである。

1,popenで作られたプロセスの終了を待つ。
2,引数で取得したfile pointerをcloseする。
3,pcloseの戻り値としてwait4で返されたコマンドの終了状態を返すようにする。

これを実現するためにライブラリ側でforkしたプロセスのpidを保持し、popen側でpidの登録をしなければならない。
そのためのコード追加と、pidの登録、削除（兼取り出し）のための関数、regist_processとremove_processを追加した。

ライブラリ側で状態を保持すること、fd関連のリソース管理、プロセスのリソース管理を行う必要がある。
pcloseはfdのcloseだけでなく、その裏で動いているプロセスの処理も行っている。
