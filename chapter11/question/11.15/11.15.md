# プログラミング課題11.15
## 課題内容
**時報サーバーのプロクシ** プロクシとは、リクエストを受付、他のサーバーにそれを転送し、
そのサーバーから送り返されてきたレスポンスをあなたに返してくるプログラムのことである。
クリーニング店の店頭と同様に、そこでクリーニングを行うわけではない。
作業場に衣類を送り、また作業場から衣類を送り返して貰うだけだ。
時報サーバーのプロクシを書きなさい。
プロクシプログラムは、標準ポートへの接続を受付、"本物の"時報サーバーへの接続を開設し、
そのサーバから時刻情報を取り出してきて、クライアントにレスポンスを送り返すものとする。

## 設計案
このサーバーで、リクエストを受けたら、サーバー側がクライアントとなって、実際のサーバーに接続し、
実際のサーバーからのレスポンスをそのままレスポンスとしてクライアントに返せばいいだけだ。
サーバー情報は以下の通り。
本物の時刻サーバー 192.168.0.9:13000
プロキシサーバー 192.168.0.7:12000
本物の時刻サーバーは接続するだけでレスポンスを返してくれる。
それを受け取りながらwriteでクライアントに送るだけで実現できるはずだ。

### コード
```

#include <netdb.h>
#include <netinet/in.h>
#include <stdio.h>
#include <string.h>
#include <sys/socket.h>
#include <unistd.h>

#define TIMESERVPORT 13000
#define PORTNUM 12000
#define HOSTLEN 256

#define oops(msg)                                                              \
  {                                                                            \
    perror(msg);                                                               \
    exit(1);                                                                   \
  }

int setup_server(int port);
int setup_cliant(char *ipaddr, int port);
ssize_t write_all(int sock_fd, char *buf, int rn);

int main(int ac, char *av[]) {
  int sock_id = setup_server(PORTNUM);
  int sock_fd = -1;

  while (1) {
    sock_fd = accept(sock_id, NULL, NULL);
    if (sock_fd == -1) {
      perror("accept");
      continue;
    }

    int cliant_sockfd = -1;
    char buf[BUFSIZ];
    if ((cliant_sockfd = setup_cliant("192.168.0.9", TIMESERVPORT)) < 0) {
      perror("setup_cliant");
      close(sock_fd);
      continue;
    }

    ssize_t rn = -1;
    ssize_t wn = -1;
    while ((rn = read(cliant_sockfd, buf, sizeof(buf))) > 0) {
      wn = write_all(sock_fd, buf, rn);
      if (wn == -1) {
        perror("write_all");
        break;
      }
    }

    if (rn == -1) {
      perror("read");
    }
    if (wn == -1) {
      perror("write_all");
    }

    close(cliant_sockfd);
    close(sock_fd);
  }

  return 0;
}

int setup_cliant(char *ipaddr, int port)
/*
 * クライアントとして第１引数で入力したサーバーに接続する。
 * 戻り値:成功時はsock_idを、失敗時は-1を返す
 */
{

  struct sockaddr_in servadd;
  char hostname[HOSTLEN];
  struct hostent *hp;
  int sock_id = -1;

  /* ステップ１：　ソケットを作成する*/
  if ((sock_id = socket(PF_INET, SOCK_STREAM, 0)) == -1) {
    return sock_id;
  }

  /** ステップ2 : サーバーに接続する*/

  memset(&servadd, 0, sizeof(servadd));
  hp = gethostbyname(ipaddr);
  if (hp == NULL) {
    perror("gethostbyname");
    return -1;
  }

  memcpy((struct sockaddr *)&servadd.sin_addr, hp->h_addr, hp->h_length);
  servadd.sin_port = htons(port);
  servadd.sin_family = AF_INET;

  if (connect(sock_id, (struct sockaddr *)&servadd, sizeof(servadd)) != 0) {
    perror("connect");
    return -1;
  }

  return sock_id;
}

int setup_server(int port) {
  struct sockaddr_in saddr;
  char hostname[HOSTLEN];
  int sock_id;

  /* ステップ1:カーネルにソケットを要求する。*/

  sock_id = socket(PF_INET, SOCK_STREAM, 0); // ソケット取得
  if (sock_id == -1)
    return -1;

  /* ステップ２：ソケットにアドレスをバインドする。アドレスはホスト、ポート*/

  memset((void *)&saddr, 0, sizeof(saddr)); /*構造体をクリアする*/

  saddr.sin_addr.s_addr = htonl(INADDR_ANY);
  saddr.sin_port = htons(
      PORTNUM); // htons()関数でリトルエンディアン→ビッグエンディアンに変換する。（ネットワークはビッグエンディアンが共通語になっている。）
  saddr.sin_family = AF_INET;

  if (bind(sock_id, (struct sockaddr *)&saddr, sizeof(saddr)) != 0)
    return -1;

  /*
   * ステップ3:キューサイズを１としてソケットを着信できる状態にする。
   */

  if (listen(sock_id, 5) != 0)
    return -1;

  return sock_id;
}

ssize_t write_all(int sock_fd, char *buf, int rn) {
  int wr = -1;
  int total = 0;
  while (total < rn) {
    wr = write(sock_fd, buf + total, rn - total);

    if (wr <= 0) {
      return -1;
    }
    total += wr;
  }
  return total;
}


```

### 実行結果(cliant側)
```
$ telnet 192.168.0.7 12000
Trying 192.168.0.7
Connected to 192.168.0.7
Escape character is '^]'.
The time here is ...Sun Apr 12 02:36:04 2026
Connection closed by foreign host.
```

クライアントは192.168.0.7の12000に接続しているにも関わらず、192.168.0.9:13000に接続したときの時報サーバーの内容と同じものが表示されている。
これはきちんと192.168.0.7がプロキシサーバーとして機能している証拠である。

## まとめ
proxyサーバーはクライアントとして、本物のサーバーに接続してレスポンスを受取り
それぞれと独立したTCP接続を確立し受信したバイト列をそのまま別の接続へ転送することで機能する。

### 注意点
proxyサーバーの情報は目標のサーバーから受け取った情報をそのままクライアントに流すだけなので
単純にバイトストリームを流すために readでそのまま目標のサーバーから読み、writeでそのままクライアントに送るのがよい。
fgetsなどは使う必要はない。
その際readの戻り値でバイト数を取得し、そのバイト数だけwriteすればよい。
readの戻り値はEOFは0、エラーは-1なので、while((len = read(...)) > 0)とすれば良い。

また、writeは一度の呼び出しで全てのデータを書き込むとは限らないため、
ループして全てのバイトを書き込む必要がある。
