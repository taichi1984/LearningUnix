# 研究課題11.11
## 課題内容
**双方向パイプ** SystemV から発展したバージョンは、双方向パイプを提供している。
次のプログラムを実行すれば、手元のバージョンが双方向パイプをサポートしているかどうかをテストできる。

```
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
/*
 *  testbdp.c - 双方向パイプをテストする。
 *
 */

int main() {
  int p[2];

  if (pipe(p) == -1)
    exit(1);
  if (write(p[0], "hello", 5) == -1)
    perror("write into pipe[0] failed");
  else
    printf("write into pipe[0] worked\n");
}
```

内部的には、双方向パイプは２本の待ち行列を持っており、片方はpipe[0]からpipe[1]に、もう片方は逆に向いている。
パイプの片方にデータを書き込むと、其のデータは反対側に向かう待ち行列に追加され、パイプの片方からデータを読み出すと、反対側から向かってくる待ち行列からデータを取り出す。
あなたのシステムが双方向パイプをサポートしていない場合、次の呼び出しを使ってパイプの対を作ることができる。

```
#include <sys/socket.h>
#include <sys/types.h>

int apipe[2];
socketpair(AF_UNIX, SOCK_STREAM, PF_UNSPEC, apipe);
```

2本の片方向パイプではなく１本の双方向パイプを使うように、tinybc.cを書き換えなさい。

## まずは自分のシステムが双方向パイプをサポートしているかどうかを確認する。
確認のためのコードをそのまま実行したところ、以下のように表示された。

```
$ gcc testbpd.c -o testbpd
$ ./testbpd
write into pipe[0] failed: Bad file descriptor
```

双方向パイプはサポートしていないようだ。
読み込み側のfdに書き込みをすることはできない。

## まずはsocketpairの使い方を実際のコードで確認する

親プロセスが入力した内容を子プロセスが標準出力に出力するだけの簡単なコードを書く。

### 
```
#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <unistd.h>

int main() {

  int apipe[2];
  if (socketpair(AF_UNIX, SOCK_STREAM, PF_UNSPEC, apipe) == -1) {
    perror("socketpair");
    exit(1);
  }

  int pid = -1;

  if ((pid = fork()) == -1) {
    perror("fork");
    exit(1);
  }

  if (pid == 0) {
    close(apipe[1]); // 親用FDは閉じる
    char buf[BUFSIZ];
    int n = 0;

    while ((n = read(apipe[0], buf, sizeof(buf))) > 0) {
      write(STDOUT_FILENO, buf, n);
    }

  } else {
    close(apipe[0]); // 子用のFDは閉じる
    int c = 0;
    while ((c = getchar()) != EOF) {
      write(apipe[1], &c, 1);
    }
  }
}
```

### 実行結果
```
$ ./test_socketpair 
aaa
aaa
aaaa
aaaa
ccc
ccc
cca
cca
```

うまくいっている。双方向パイプを使う場合は、fork,execした後にどうfdを閉じるかが重要なようだ。


## tinybcを双方向パイプ版で書きなおす。
### ソース
```
/** tinybc.c * dcを使って仕事をする簡単な電卓
 *　　　　　　* 双方向パイプの作り方、使い方を示す。
 *            * 入力は、数値、演算子、数値の順番で指定する。
 *            * tinybcは、これを数値、\n、数値、\n,演算子,
 *            \n , p に変換してdcに渡す。
 *            * 返された結果は、標準出力に書き込む。
 *
 *            *プログラムの概要
 *            *a. socketpairで双方向パイプを作る。
 *            *b. fork呼び出し（新しいプロセスを得る）
 *            *c.
 *dcになる予定の新プロセスの中で、標準入出力をパイプに接続し、execl dcする。 *d.
 *tinybcプロセスの中では、特に配管の必要はない。
 *
 *            通常の入出力を介して、ユーザーとやり取りし、
 *            パイプを介して、プログラム感の情報のやり取りをする。
 *            最後にパイプをクローズすると、dcは終了する。
 *            *注意:複数行のレスポンスを処理しない。
 *
 */

#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <sys/wait.h>
#include <unistd.h>
#define oops(m, x)                                                             \
  {                                                                            \
    perror(m);                                                                 \
    exit(x);                                                                   \
  }
void be_dc(int apipe[2]);
void be_bc(int apipe[2]);
void fatal(char *mess);

int main() {
  int pid = -1;
  int apipe[2];

  /* socketpairを作る */

  if (socketpair(AF_UNIX, SOCK_STREAM, PF_UNSPEC, apipe) == -1) {
    perror("socketpair");
    exit(1);
  }

  /* ユーザーインターフェースのためのプロセスを作る */

  if ((pid = fork()) == -1)
    oops("cannot fork", 2);

  if (pid == 0)
    be_dc(apipe);
  else {
    be_bc(apipe);
    wait(NULL);
  }
}

void be_dc(int apipe[2])
/*
 * 標準入出力をセットアップしてから、最後にexecl dc
 */
{
  close(apipe[0]);
  /* 標準入力を子用パイプに接続する*/
  if (dup2(apipe[1], 0) == -1)
    oops("dc: cannot redirect stdin", 3);

  /*標準出力を子用パイプに接続する*/
  if (dup2(apipe[1], 1) == -1)
    oops("dc:cannot redirect stdout", 4);

  close(apipe[1]);
  /* -オプションをつけてdcをexeclにする */
  execlp("dc", "dc", "-", NULL);
  oops("Cannot run dc", 5);
}

void be_bc(int apipe[2])
/*
 * 標準入力から読み出したものを逆ポーランド記法に変換し、パイプに送る。
 * 同じパイプから処理結果を読み出し、ユーザーにしめす。
 * fdopen()を呼び出して、ファイルディスクリプタをストリームに変換する。
 */
{
  int num1, num2;
  char operation[BUFSIZ], message[BUFSIZ];
  FILE *fp;

  /* セットアップ */
  close(apipe[1]);

  fp = fdopen(apipe[0], "r+");

  if (fp == NULL)
    fatal("Error convering pipes to streams");

  /* メインループ */
  while (1) {

    printf("tinybc : ");
    if (fgets(message, BUFSIZ, stdin) == NULL)
      break;

    /* parse input */
    if (sscanf(message, "%d%1[-+*/^]%d", &num1, operation, &num2) != 3) {
      printf("syntax error\n");
      continue;
    }

    if (fprintf(fp, "%d\n%d\n%c\np\n", num1, num2, *operation) == EOF)

      fatal("Error writing");

    fflush(fp);

    if (fgets(message, BUFSIZ, fp) == NULL)
      break;

    printf("%d %c %d = %s", num1, *operation, num2, message);
  }
  fclose(fp);
}

void fatal(char *mess) {
  fprintf(stderr, "Error: %s\n", mess);
  exit(1);
}


```
### 実行結果
```
$ ./tinybc_duplex_pipe
tinybc : 2+2
2 + 2 = 4
```

## まとめ
socketpairは双方向通信を1つのFD対で実現できるが、
通信の責務を明確にするために、各プロセスが一方のFDのみを使用する設計とした。

今回のケースでは親と子で双方向パイプを使うときは、一つのfdを親用、１つのfdを子用として使いわかりやすく実装した。

dup2を使い、一つのfdを標準入力と標準出力の２つにつなぐことなどは可能。
その際は入力と出力が同じfdに来ることになるので、読み書きが可能なfdでないと成立しない。
