# 研究課題11.12
## 課題内容
**IPのブロック** 特定のIPアドレスを持つマシンのクライアントからのリクエストのみに応答するように、timeserv.cを書き換えなさい。
サーバは、呼び出しを受付、クライアントのアドレスをチェックする。クライアントが指定されたIPアドレスで実行されていなければ、サーバーは接続を切断する。
そうでなければ、サーバーは時刻メッセージを送り返す。
さらに、サーバーが受け付けられるIPアドレスのリストをファイルから読み出すように、このIPブロック機能を拡張しなさい。
また、このテクニックの実用的な応用方法を説明しなさい。

## どうやって接続を切断するか。
基本的にはaccept時に返されたfdをcloseすれば接続は切断できるので、
accept時に取得した情報を元にリストと照合して、一致してなければ何も処理せずにcloseしてしまえばよい。
auth_ip()という真偽値を返す関数を作って照合すればいいだろう。

### コード
```

#include <arpa/inet.h>
#include <netdb.h>
#include <netinet/in.h>
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <strings.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <time.h>
#include <unistd.h>

#define PORTNUM 13000
#define HOSTLEN 256
#define oops(msg)                                                              \
  {                                                                            \
    perror(msg);                                                               \
    exit(1);                                                                   \
  }
int setup_server(int port);
bool auth_ip(char *ip);

int main(int ac, char *av[]) {
  struct sockaddr_in saddr_cliant;
  socklen_t saddr_cliantlen;
  int sock_id, sock_fd;
  FILE *sock_fp;
  time_t thetime;

  sock_id = setup_server(PORTNUM);
  if (sock_id == -1) {
    perror("setup_server");
    exit(1);
  }
  while (1) {

    memset((void *)&saddr_cliant, 0,
           sizeof(saddr_cliant)); /*構造体をクリアする*/

    char ip[INET_ADDRSTRLEN];
    saddr_cliantlen = sizeof(saddr_cliant);

    sock_fd =
        accept(sock_id, (struct sockaddr *)&saddr_cliant, &saddr_cliantlen);

    if (sock_fd == -1)
      oops("accept");

    inet_ntop(AF_INET, &saddr_cliant.sin_addr, ip, sizeof(ip));

    char host[NI_MAXHOST];

    int ret = getnameinfo((struct sockaddr *)&saddr_cliant, saddr_cliantlen,
                          host, sizeof(host), 0, 0, 0);

    if (!auth_ip(ip)) {
      close(sock_fd);
      printf("%s からの接続を切断しました。\n", ip);
      continue;
    };

    printf("Got a call ");
    if (ret == 0) {
      printf("from %s (%s)\n", ip, host);
    } else {
      printf("from %s \n", ip);
    }

    sock_fp = fdopen(sock_fd, "w");

    if (sock_fp == NULL)
      oops("fdopen");

    thetime = time(NULL);

    fprintf(sock_fp, "The time here is ...");
    fprintf(sock_fp, "%s", ctime(&thetime));
    fclose(sock_fp);
  }
}

int setup_server(int port) {
  struct sockaddr_in saddr;
  char hostname[HOSTLEN];
  int sock_id;

  /* ステップ1:カーネルにソケットを要求する。*/

  sock_id = socket(PF_INET, SOCK_STREAM, 0); // ソケット取得
  if (sock_id == -1)
    return -1;

  /* ステップ２：ソケットにアドレスをバインドする。アドレスはホスト、ポート*/

  memset((void *)&saddr, 0, sizeof(saddr)); /*構造体をクリアする*/

  saddr.sin_addr.s_addr = htonl(INADDR_ANY);
  saddr.sin_port = htons(
      PORTNUM); // htons()関数でリトルエンディアン→ビッグエンディアンに変換する。（ネットワークはビッグエンディアンが共通語になっている。）
  saddr.sin_family = AF_INET;

  if (bind(sock_id, (struct sockaddr *)&saddr, sizeof(saddr)) != 0)
    return -1;

  /*
   * ステップ3:キューサイズを１としてソケットを着信できる状態にする。
   */

  if (listen(sock_id, 1) != 0)
    return -1;

  return sock_id;
}

bool auth_ip(char *ip) {
  if (strcmp(ip, "127.0.0.1") == 0) {
    return false;
  }
  return true;
}
``` 

auth_ipで127.0.0.1だったら確認して切断するように設計した。

## 接続可能なリストからのみの接続を受け入れるようにする。
接続可能なリストからの接続のみを受け入れるようにするには、ipのリストを１行ずつ読み、
そのipがリストに乗ってた場合のみauth_ipがtrueを返すようにすればいい。
但し今回はauth_ipがiplistを読むようになるため、ファイルの接続に失敗する可能性がある。
これを考慮し、戻り値をbool値から以下のように変更する。

1:認証成功
0:認証失敗
-1:エラー発生

listはtimeserv.cがおいてあるフォルダにiplistというファイルにして置くことにする。

### コード

```

#include <netdb.h>
#include <netinet/in.h>
#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <strings.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <time.h>
#include <unistd.h>

    exit(1);                                                                   \
  }

int auth_ip(char *ip);

int main(int ac, char *av[]) {
  struct sockaddr_in saddr_cliant;
  socklen_t saddr_cliantlen;
  int sock_id, sock_fd;
  FILE *sock_fp;
  time_t thetime;

  sock_id = setup_server(PORTNUM);
  if (sock_id == -1) {
    perror("setup_server");
    exit(1);
  }
  while (1) {

    memset((void *)&saddr_cliant, 0,
           sizeof(saddr_cliant)); /*構造体をクリアする*/

    char ip[INET_ADDRSTRLEN];
    saddr_cliantlen = sizeof(saddr_cliant);

    sock_fd =
        accept(sock_id, (struct sockaddr *)&saddr_cliant, &saddr_cliantlen);

    if (sock_fd == -1)
      oops("accept");

    inet_ntop(AF_INET, &saddr_cliant.sin_addr, ip, sizeof(ip));

    char host[NI_MAXHOST];

    int ret = getnameinfo((struct sockaddr *)&saddr_cliant, saddr_cliantlen,
                          host, sizeof(host), 0, 0, 0);

    int auth_ret = -1;
    if ((auth_ret = auth_ip(ip)) == -1) {
      perror("auth_ip");
      close(sock_fd);
      continue;
    } else if (auth_ret == 0) {
      close(sock_fd);
      printf("%s からの接続を切断しました。\n", ip);
      continue;
    }

    printf("Got a call ");
    if (ret == 0) {
      printf("from %s (%s)\n", ip, host);
    } else {
      printf("from %s \n", ip);
    }

    sock_fp = fdopen(sock_fd, "w");

    if (sock_fp == NULL)
      oops("fdopen");

    thetime = time(NULL);

    fprintf(sock_fp, "The time here is ...");
    fprintf(sock_fp, "%s", ctime(&thetime));
    fclose(sock_fp);
  }
}

int setup_server(int port) {
  struct sockaddr_in saddr;
  char hostname[HOSTLEN];
  int sock_id;

  /* ステップ1:カーネルにソケットを要求する。*/

  sock_id = socket(PF_INET, SOCK_STREAM, 0); // ソケット取得
  if (sock_id == -1)
    return -1;

  /* ステップ２：ソケットにアドレスをバインドする。アドレスはホスト、ポート*/

  memset((void *)&saddr, 0, sizeof(saddr)); /*構造体をクリアする*/

  saddr.sin_addr.s_addr = htonl(INADDR_ANY);
  saddr.sin_port = htons(
      PORTNUM); // htons()関数でリトルエンディアン→ビッグエンディアンに変換する。（ネットワークはビッグエンディアンが共通語になっている。）
  saddr.sin_family = AF_INET;

  if (bind(sock_id, (struct sockaddr *)&saddr, sizeof(saddr)) != 0)
    return -1;

  /*
   * ステップ3:キューサイズを１としてソケットを着信できる状態にする。
   */

  if (listen(sock_id, 1) != 0)
    return -1;

  return sock_id;
}

int auth_ip(char *ip)
/*
 *  iplistから接続可能なipを読み込み、今接続しているホストのipと照合する。
 *  接続可能であれば1を返す。
 *  接続不可であれば0を返す。
 *  エラー発生時は-1を返す。
 */
{
  FILE *fp;
  char buf[INET_ADDRSTRLEN];
  fp = fopen("iplist", "r");

  if (fp == NULL) {
    perror("fopen");
    return -1;
  }

  while (fgets(buf, sizeof(buf), fp)) {
    buf[strcspn(buf, "\n")] = '\0';
    if (strcmp(ip, buf) == 0) {
      printf("認証できました\n");
      fclose(fp);
      return 1;
    }
  }
  printf("認証できませんでした\n");
  fclose(fp);
  return 0;
}
```

## まとめ
### 切断の方法について
acceptで取得したクライアントの情報を元にinet_ntop関数でクライアントのipを取得して、
そのipをリストと照合することによって、ipごとに接続後の処理を分岐できる。
今回のケースでは接続後に、クライアントipがipリスト内に含まれていない場合に切断することにより
ipによる認証が失敗したクライアントを拒否することができた。

### 実装時の注意
照合時にfgetsでiplistから行を取得することにより、"\n"が入るため、ipとの照合が常に失敗することがあった。
fgetsで取得できる文字列では改行文字も含まれることに注意が必要。
また、関数の戻り値としては成否判断をするだけであっても、実際はエラーのケースもあるので、
bool値を返す関数は設計的に好ましくないケースも多くある。

### 実用的な応用方法について
このiplistの内容は接続可能なホストを表すものだが、これを接続不可能なホストにしてしまえば、
それはアクセス禁止リストになる。迷惑なユーザーや悪質なユーザーをbanするための手段として有効になる。
