# プログラミング演習 3.19  
## 課題内容  
**アクセス日時の表示** 標準のlsは、最終変更日時ではなく、最終アクセス日時を表示する-uオプションをサポートしている。  
-lを指定せずに-uを指定するとどうなるかを答えなさい。また、-uをサポートするようにls2.cを書き換えなさい。  
-lを指定しなくても、-uを指定できるようにすることにはどのような意味があるか答えなさい。  
ヒント：-tオプションの説明を参照すること。  
  
## -lを指定して-uを指定するとどうなるか?  
```  
$ ls -l -u  
合計 48  
-rw-r--r-- 1 taichi84 taichi84  7454  8月 31 23:30 3.18.md  
drwxr-xr-x 3 taichi84 taichi84  4096  9月  1 22:01 direct1  
drwxr-xr-x 2 taichi84 taichi84  4096  9月  1 22:01 direct2  
drwxr-xr-x 2 taichi84 taichi84  4096  9月  1 22:01 direct3  
-rwxr-xr-x 1 taichi84 taichi84 17024  8月 31 23:25 ls2  
-rw-r--r-- 1 taichi84 taichi84  4589  8月 31 23:25 ls2.c  
```  
  
stat等で確認した結果、通常の時間ではなくaccess_timeが表示されるようになっている。  
(上記の実行結果では、lsにはヘッダがない関係でわかりにくい)  
  
## -lを指定せずに-uを指定するとどうなるか？  
  
```  
$ ls -u  
direct1  direct2  direct3  3.18.md  ls2  ls2.c  
```  
  
ファイルの並び順がaccess_time順にソートされる。

```  
/*

{  
  DIR *dir_ptr;  
  struct dirent *direntp;

  struct stat entry;  
  
  if ((dir_ptr = opendir(dirname)) == NULL)  
    fprintf(stderr, "ls1: cannot open %s \n", dirname);  
  else {  
    printf("%s :\n", dirname);  
    while ((direntp = readdir(dir_ptr)) != NULL) {  
      if (direntp->d_name[0] == '.')  
        continue;  
      strcpy(buf, dirname);  
      strcat(buf, "/");  
      strcat(buf, direntp->d_name);  
  
      if (option.show_long) {  
        showstats(buf, option);  
      } else {  
        printf("%s ", direntp->d_name);  
      }  
    }  
    printf("\n\n");  
    seekdir(dir_ptr, SEEK_SET);  
  
    if (option.recursion) {  
      while ((direntp = readdir(dir_ptr)) != NULL) {  
        if (direntp->d_name[0] == '.')  
          continue;  
        if (direntp->d_type == DT_DIR) {  
          char recdir[257];  
          snprintf(recdir, sizeof(recdir), "%s/%s", dirname, direntp->d_name);  
          do_ls(recdir, option);  
        }  
      }  
    }  
  }  
  closedir(dir_ptr);  
}

}
```  

### 改善案第２フェーズ  
-l == 0 かつ -uの場合に、表示順の並び替えをする必要がある。  
比較関数とqsortを使えば、上手くいくとは思うが、どうやって表示順を制御するか？  
一度配列に格納する形ではなく、表示順を変更できるのかを考える必要がある。  
  
この答えはlsをソートした課題、3.15にあると思われるので参照し、改良を加えた。  
  
```  
  
/** ls2.c  
 *  目的 : ディレクトリ( 複数の場合も含む) の内容をリストアップする。 ls-l仕様  
 *  動作 :  
 * 引数がない場合には、"."、そうでなければ引数のディレクトリに含まれるファイルを出力する。  
 *  
 */  
  
#define _DEFAULT_SOURCE  
#include <assert.h>  
#include <dirent.h> // DIR*, opendir/readdir（使っていれば）  
#include <grp.h>    // getgrgid  
#include <pwd.h>    // getpwuid  
#include <stdio.h>  
#include <stdlib.h>  
#include <string.h>  
#include <sys/stat.h>  // struct stat  
#include <sys/types.h> // mode_t, uid_t, gid_t  
#include <time.h>      // struct timespec, localtime, strftime  
#include <unistd.h>    // close など（必要なら）  
  
typedef struct {  
  int show_long;  
  int show_all;  
  int recursion;  
  int atime;  
  int reverse;  
} Options;  
  
void showtime(struct timespec *time);  
void mode_to_letter(mode_t mode, char str[]);  
char *uid_to_name(uid_t uid);  
char *gid_to_name(gid_t gid);  
void do_ls(char[], Options);  
void showstats(char *d_name, Options);  
  
struct fileinfo {  
  char *name;  
  struct stat st;  
};  
  
int cmp_d_name(const void *a, const void *b) {  
  // a, b は "要素へのポインタへのポインタ"  
  const struct fileinfo *pa = *(const struct fileinfo **)a;  
  const struct fileinfo *pb = *(const struct fileinfo **)b;  
  return strcmp(pa->name, pb->name); // strcmp をそのまま使える  
}  
  
int cmp_d_name_desc(const void *a, const void *b) { return -cmp_d_name(a, b); }  
  
int cmp_atime(const void *a, const void *b) {  
  // a, b は "要素へのポインタへのポインタ"  
  const struct fileinfo *pa = *(const struct fileinfo **)a;  
  const struct fileinfo *pb = *(const struct fileinfo **)b;  
  
  if (pa->st.st_atim.tv_sec != pb->st.st_atim.tv_sec)  
    return (pa->st.st_atim.tv_sec > pb->st.st_atim.tv_sec) -  
           (pa->st.st_atim.tv_sec < pb->st.st_atim.tv_sec);  
  
  return (pa->st.st_atim.tv_nsec > pb->st.st_atim.tv_nsec) -  
         (pa->st.st_atim.tv_nsec < pb->st.st_atim.tv_nsec);  
}  
  
int cmp_atime_desc(const void *a, const void *b) { return -cmp_atime(a, b); }  
  
int main(int ac, char *av[]) {  
  
  Options option = {0, 0, 0, 0, 0};  
  int file_argc = 0;  
  if (ac == 1) {  
    do_ls(".", option);  
  } else {  
    for (int i = 1; i < ac; i++) {  
      if (av[i][0] == '-') {  
        if (strcmp(av[i], "-R") == 0) {  
          option.recursion = 1;  
        } else if (strcmp(av[i], "-l") == 0) {  
          option.show_long = 1;  
        } else if (strcmp(av[i], "-u") == 0) {  
          option.atime = 1;  
        } else if (strcmp(av[i], "-r") == 0) {  
          option.reverse = 1;  
        } else {  
          fprintf(stderr, "Unknown option: %s\n", av[i]);  
        }  
      } else {  
        do_ls(av[i], option);  
        file_argc++;  
      }  
    }  
    if (file_argc == 0) {  
  
      do_ls(".", option);  
    }  
    return 0;  
  }  
}  
  
void do_ls(char dirname[], Options option)  
/*  
 *  dirnameというディレクトリのファイルをリストアップする。  
 */  
{  
  
  DIR *dir_ptr;  
  struct dirent *direntp;  
  struct fileinfo *fileinfo_array[1024];  
  int i = 0;  
  char buf[512];  
  
  printf(": %s \n", dirname);  
  if ((dir_ptr = opendir(dirname)) == NULL)  
    fprintf(stderr, "ls1: cannot open %s \n", dirname);  
  else {  
    while ((direntp = readdir(dir_ptr)) != NULL) {  
      fileinfo_array[i] = malloc(sizeof(struct fileinfo));  
      fileinfo_array[i]->name = strdup(direntp->d_name);  
      snprintf(buf, sizeof(buf), "%s/%s", dirname, fileinfo_array[i]->name);  
      stat(buf, &fileinfo_array[i]->st);  
      i++;  
    }  
  }  
  
  if (option.atime == 1 && option.reverse == 1) {  
    qsort(fileinfo_array, i, sizeof(fileinfo_array[0]), cmp_atime_desc);  
  } else if (option.atime == 1) {  
    qsort(fileinfo_array, i, sizeof(fileinfo_array[0]), cmp_atime);  
  } else if (option.reverse == 1) {  
    qsort(fileinfo_array, i, sizeof(fileinfo_array[0]), cmp_d_name_desc);  
  } else if (option.reverse == 0) {  
    qsort(fileinfo_array, i, sizeof(fileinfo_array[0]), cmp_d_name);  
  }  
  
  for (int j = 0; j < i; j++) {  
    if (fileinfo_array[j]->name[0] == '.')  
      continue;  
  
    snprintf(buf, sizeof(buf), "%s/%s", dirname, fileinfo_array[j]->name);  
  
    if (option.show_long)  
      showstats(buf, option);  
    else  
      printf("%10s ", fileinfo_array[j]->name);  
  }  
  printf("\n\n");  
  
  if (option.recursion) {  
    for (int j = 0; j < i; j++) {  
      if (fileinfo_array[j]->name[0] == '.')  
        continue;  
  
      if (S_ISDIR(fileinfo_array[j]->st.st_mode)) {  
        char recdir[257];  
        snprintf(recdir, sizeof(recdir), "%s/%s", dirname,  
                 fileinfo_array[j]->name);  
        do_ls(recdir, option);  
      }  
    }  
  }  
  closedir(dir_ptr);  
  for (int j = 0; j < i; j++) {  
    free(fileinfo_array[j]->name);  
    free(fileinfo_array[j]);  
  }  
}

  struct stat *bufp = &buf;

    printf("%10s ", gid_to_name(bufp->st_gid));  
    printf("%10ld bytes   ", bufp->st_size);  
    if (option.atime == 1)  
      showtime(&bufp->st_atim);  
    else  
      showtime(&bufp->st_mtim);  
    printf("%20s ", strrchr(filename, '/') + 1);  
    printf("\n");

}
```  

### ついでの修正  
データの取得処理をわけたことで再帰処理時に再度readdirを行わなくてもよくなった。  
これでシステムコールを呼ぶ回数が減り効率もよくなった。  
  
  
  
## -lを指定しなくても-uを指定できるようにすることにはどのような意味があるのか？  
まず、表示がなくてもアクセス日時順に昇順降順にして並べてファイル名データとして扱うことが可能。  
  
$ ./ls2 -u > textfile  
などでアクセス日時順にならんだファイル名を出力することができる。  
これは配列データみたいなものが欲しいときに使える仕組みになっていると思う。  
データソースとしてlsを使う時には有効な機能になっている。
