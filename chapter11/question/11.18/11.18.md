# プログラミング課題11.18
## 課題内容
**パン屋の番号サーバー** 一部のパン屋さんには、お客さんに番号札を渡す機械がある。
"販売中"と書かれている上には、次に受け付けてもらえるお客さんの番号が表示される。
このパン屋の番号札サービスを実装するクライアントとサーバーを書きなさい。
サーバーは連番を発行すること。ユーザーは、クライアントプログラムを実行して、サーバーから番号を取得するものとする。 ## 設計イメージ クライアント側はプログラムを実行する（ボタンを押すイメージ）でサーバーから番号を取得できる。
サーバーはクライアントからのアクセスを検知して番号をクライアントに渡す。
クライアント側はそれを表示すれば良い。

#include <netdb.h>

int setup_server(int port);

int main() {
  int sock_id = setup_server(PORTNUM);
  int que_no = 1;
  ssize_t wn = -1;
  while (1) {
    int sock_fd = accept(sock_id, NULL, NULL);
    if (sock_fd == -1) {
      perror("accept");
      continue;
    }
    int net_no = htonl(que_no);
    wn = write_all(sock_fd, &net_no, sizeof(int));
    if (wn == -1) {

      continue;
    }

    que_no++;
    close(sock_fd);
  }
}

int setup_server(int port) {
  struct sockaddr_in saddr;
  char hostname[HOSTLEN];
  int sock_id;

  /* ステップ1:カーネルにソケットを要求する。*/

  sock_id = socket(PF_INET, SOCK_STREAM, 0); // ソケット取得
  if (sock_id == -1)
    return -1;

  /* ステップ２：ソケットにアドレスをバインドする。アドレスはホスト、ポート*/

  memset((void *)&saddr, 0, sizeof(saddr)); /*構造体をクリアする*/

  saddr.sin_addr.s_addr = htonl(INADDR_ANY);
  saddr.sin_port = htons(
      port); // htons()関数でリトルエンディアン→ビッグエンディアンに変換する。（ネットワークはビッグエンディアンが共通語になっている。）
  saddr.sin_family = AF_INET;

  if (bind(sock_id, (struct sockaddr *)&saddr, sizeof(saddr)) != 0)
    return -1;

  /*
   * ステップ3:キューサイズを１としてソケットを着信できる状態にする。
   */

  if (listen(sock_id, 5) != 0)
    return -1;

  return sock_id;
}

ssize_t write_all(int sock_fd, const void *buf, size_t rn) {
  ssize_t wr = -1;
  ssize_t total = 0;
  const char *p = buf;

  while (total < rn) {
    wr = write(sock_fd, p + total, rn - total);

    if (wr < 0) {
      return -1;
    }
    if (wr == 0) {
      break;
    }
    total += wr;
  }
  return total;
}

int setup_client(char *ipaddr, int port)
/*
 * クライアントとして第１引数で入力したサーバーに接続する。
 * 戻り値:成功時はsock_idを、失敗時は-1を返す
 */
{

  struct sockaddr_in servadd;
  char hostname[HOSTLEN];
  struct hostent *hp;
  int sock_id = -1;

  /* ステップ１：　ソケットを作成する*/
  if ((sock_id = socket(PF_INET, SOCK_STREAM, 0)) == -1) {
    return sock_id;
  }

  /** ステップ2 : サーバーに接続する*/

  memset(&servadd, 0, sizeof(servadd));
  hp = gethostbyname(ipaddr);
  if (hp == NULL) {
    perror("gethostbyname");
    return -1;
  }

  memcpy((struct sockaddr *)&servadd.sin_addr, hp->h_addr, hp->h_length);
  servadd.sin_port = htons(port);
  servadd.sin_family = AF_INET;

  if (connect(sock_id, (struct sockaddr *)&servadd, sizeof(servadd)) != 0) {
    perror("connect");
    return -1;
  }

  return sock_id;
}

ssize_t read_all(int fd, void *buf, size_t len) {
  ssize_t total = 0;
  ssize_t rn;
  char *p = buf;
  while (total < len) {
    rn = read(fd, p + total, len - total);
    if (rn < 0) {
      return -1;
    }
    if (rn == 0) {
      break;
    }
    total += rn;
  }

  return total;
}

### まとめ
今回はサーバーとクライアント側の処理を両方書く課題だったが、
サーバー側では今の番号を変数に保持することと、通信面では文字列以外の数字を送る必要があった。
readとwriteの第２引数はvoid型のポインタなので要は欲しい物は送りたいデータの先頭アドレスの値である。
int型で送りたいデータは解釈が必要なデータのため、サーバー側はhtonl関数、クライアント側はntohl関数を使って通信時のデータを揃えてやる必要がある。

送るバイト数は第３引数できまるので、sizeof(int)か、第２引数に指定した変数のsizeを指定してやればよい。

あとはサーバー側はエラー時には番号をインクリメントせずに開いたリソースを閉じてcontinue。
クライアント側は接続したらreadして閉じればいいだけだ。
エラーチェックに関しては、クライアント側は1回行ったら終了するプログラムなので、エラー時は素直にexit(1)すればよい。
