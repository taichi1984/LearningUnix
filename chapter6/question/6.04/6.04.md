# プログラミング課題6.4
## 課題内容
play_againにおけるシグナル処理

(a) キーボードからのシグナルを無視し、イエスカノーの入力だけにおうとうするように、play_again3.cを書き換えなさい。

(b) キーボードからのシグナルを受け取ったら、端末属性をリセットし、値２を返して終了するようにplay_again3.cを書き換えなさい。

## play_again3.cの内容

```
#include <ctype.h>
#include <fcntl.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <termios.h>
#include <unistd.h>

#define ASK "Do you want another transaction?"
#define TRIES 3     // 試行回数の上限
#define SLEEPTIME 2 // １回の施行にかける時間
#define BEEP putchar('\a')

int get_response(char *, int maxtries);
void set_cr_noecho_mode();
void tty_mode(int how);
void set_nodelay_mode();
int get_ok_char();

int main() {
  int response;

  tty_mode(0);
  set_cr_noecho_mode();
  set_nodelay_mode();
  response = get_response(ASK, TRIES);
  tty_mode(1);
  return response;
}

int get_response(char *question, int maxtries) {
  int input;
  printf("%s (y/n)?", question);
  fflush(stdout);
  while (1) {
    sleep(SLEEPTIME);
    input = tolower(get_ok_char());
    if (input == 'y')
      return 0;
    if (input == 'n')
      return 1;
    if (maxtries-- == 0)
      return 2;
    BEEP;
  }
}

int get_ok_char() {
  int c;
  while ((c = getchar()) != EOF && strchr("yYnN", c) == NULL)
    ;
  return c;
}

void set_cr_noecho_mode() {
  struct termios ttystate;

  tcgetattr(0, &ttystate);
  ttystate.c_lflag &= ~ICANON;
  ttystate.c_lflag &= ~ECHO;
  ttystate.c_cc[VMIN] = 1;
  tcsetattr(0, TCSANOW, &ttystate);
}

void set_nodelay_mode() {
  int termflags;
  termflags = fcntl(0, F_GETFL);
  termflags |= O_NDELAY;
  fcntl(0, F_SETFL, termflags);
}

void tty_mode(int how) {
  static struct termios original_mode;
  static int original_flags;
  if (how == 0) {
    tcgetattr(0, &original_mode);
    original_flags = fcntl(0, F_GETFL);
  } else {
    tcsetattr(0, TCSANOW, &original_mode);
    fcntl(0, F_SETFL, original_flags);
  }
}
```
この時点ではシグナル処理はされていない。

## (a) シグナルを無視する処理を書く
```

```

キーボードからのシグナル３種を無視するignore_signal関数を書いた。
これでSIGINT,SIGTSTP,SIGQUITの３種は無視される。


## (b)キーボードからの３種のシグナルをハンドルして端末設定をもとに戻して値２を返して終了させるプログラム
```
#include <ctype.h>
#include <fcntl.h>
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <termios.h>
#include <unistd.h>

#define ASK "Do you want another transaction?"
#define TRIES 3     // 試行回数の上限
#define SLEEPTIME 2 // １回の施行にかける時間
#define BEEP putchar('\a')

int get_response(char *, int maxtries);
void set_cr_noecho_mode();
void tty_mode(int how);
void set_nodelay_mode();
int get_ok_char();
void keyboard_signal_handler(int signum);
void signal_setting();

int main() {
  int response;

  tty_mode(0);
  set_cr_noecho_mode();
  set_nodelay_mode();
  signal_setting();
  response = get_response(ASK, TRIES);
  tty_mode(1);
  return response;
}

void signal_setting() {
  signal(SIGINT, keyboard_signal_handler);
  signal(SIGQUIT, keyboard_signal_handler);
  signal(SIGTSTP, keyboard_signal_handler);
}

void keyboard_signal_handler(int signum) {
  {
    tty_mode(1);
    exit(2);
  }
}

int get_response(char *question, int maxtries) {
  int input;
  printf("%s (y/n)?", question);
  fflush(stdout);
  while (1) {
    sleep(SLEEPTIME);
    input = tolower(get_ok_char());
    if (input == 'y')
      return 0;
    if (input == 'n')
      return 1;
    if (maxtries-- == 0)
      return 2;
    BEEP;
  }
}

int get_ok_char() {
  int c;
  while ((c = getchar()) != EOF && strchr("yYnN", c) == NULL)
    ;
  return c;
}

void set_cr_noecho_mode() {
  struct termios ttystate;

  tcgetattr(0, &ttystate);
  ttystate.c_lflag &= ~ICANON;
  ttystate.c_lflag &= ~ECHO;
  ttystate.c_cc[VMIN] = 1;
  tcsetattr(0, TCSANOW, &ttystate);
}

void set_nodelay_mode() {
  int termflags;
  termflags = fcntl(0, F_GETFL);
  termflags |= O_NDELAY;
  fcntl(0, F_SETFL, termflags);
}

void tty_mode(int how) {
  static struct termios original_mode;
  static int original_flags;
  if (how == 0) {
    tcgetattr(0, &original_mode);
    original_flags = fcntl(0, F_GETFL);
  } else {
    tcsetattr(0, TCSANOW, &original_mode);
    fcntl(0, F_SETFL, original_flags);
  }
}

```
signal_setting()関数とkeyboard_signal_handlerを追加して、シグナルの制御を行うようにした。
すべてのキーボードからのシグナルに対して同じ動作を行えばいいので、keyboard_signal_handlerで、ハンドラはすべて統一した。


## 補足として
今回はキーボードから送れるシグナル３種（SIGINT,SIGTSTP,SIGQUIT)について対処したがこれで全ての強制終了に対処できたわけではない。
SIGSTOP,SIGKILLなどは捕捉できない仕様になっているため、この場合は端末設定などの初期化は出来ない。
